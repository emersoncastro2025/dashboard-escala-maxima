⚙️ Como configurar — via Google Sheets (apenas 1 vez)
1
No Google Ads, vá em Relatórios → Relatórios predefinidos → Campanhas. Adicione as colunas: Campanha, Impressões, Cliques, Custo, Conversões. Clique em Baixar → Google Sheets.
2
Na planilha aberta, copie o ID da URL: docs.google.com/spreadsheets/d/SEU_ID_AQUI/edit
3
Acesse script.google.com com crcindicarf@gmail.com, crie um Novo projeto, apague o código existente e cole o código abaixo. Substitua SEU_SHEET_ID pelo ID copiado. Salve (Ctrl+S).
4
Clique em Implantar → Nova implantação. Tipo: App da Web. Executar como: Eu. Acesso: Qualquer pessoa. Implante e autorize.
5
Copie a URL do app da web e cole no campo abaixo. Clique em Salvar & Carregar Dados. 🚀
// ⚠️ Substitua pelo ID da sua planilha do Google Ads
var SHEET_ID = 'SEU_SHEET_ID_AQUI';

function doGet(e) {
  var result;
  try {
    var ss = SpreadsheetApp.openById(SHEET_ID);
    var sheet = ss.getSheets()[0];
    var raw = sheet.getDataRange().getValues();
    if (raw.length < 2) {
      result = { ok: false, error: 'Planilha vazia ou sem dados.' };
    } else {
      var headers = raw[0].map(function(h){ return String(h).toLowerCase().trim(); });
      var find = function(keys) {
        for (var i = 0; i < headers.length; i++)
          for (var j = 0; j < keys.length; j++)
            if (headers[i].indexOf(keys[j]) >= 0) return i;
        return -1;
      };
      var cCamp = find(['campanha','campaign','nome']);
      var cImp  = find(['impress']);
      var cCli  = find(['clique','click']);
      var cCust = find(['custo','cost','gasto','valor']);
      var cConv = find(['convers']);
      var rows = [];
      for (var r = 1; r < raw.length; r++) {
        var row = raw[r];
        if (cCamp < 0 || !row[cCamp]) continue;
        var cli = parseInt(String(row[cCli]).replace(/\D/g,'')) || 0;
        var imp = parseInt(String(row[cImp]).replace(/\D/g,'')) || 0;
        var gst = parseFloat(String(row[cCust]).replace(/[^\d,\.]/g,'').replace(',','.')) || 0;
        var cnv = parseFloat(String(row[cConv]).replace(',','.')) || 0;
        rows.push({
          campanha: row[cCamp], impressoes: imp, cliques: cli, gasto: gst,
          conversoes: cnv,
          ctr: imp > 0 ? (cli/imp)*100 : 0,
          cpc: cli > 0 ? gst/cli : 0,
          custo_por_conversao: cnv > 0 ? gst/cnv : 0
        });
      }
      result = { ok: true, data: rows, ts: new Date().toISOString() };
    }
  } catch(err) {
    result = { ok: false, error: err.message };
  }
  var json = JSON.stringify(result);
  var cb = e && e.parameter && e.parameter.callback;
  if (cb) {
    return ContentService.createTextOutput(cb + '(' + json + ')')
      .setMimeType(ContentService.MimeType.JAVASCRIPT);
  }
  return ContentService.createTextOutput(json)
    .setMimeType(ContentService.MimeType.JSON);
}
